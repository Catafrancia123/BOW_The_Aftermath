# This is for modifiers that happen every day, month or year
on_actions = {
    on_monthly = {
        effect = {
            if = {
                limit = { 
                    has_idea = mod_lockdown_outer
                    check_variable = {
                        var = covid_months
                        value = 20
                        compare = less_than_or_equals
                    }
                }
                
                subtract_from_variable = { var:modifier@local_resources_factor = 0.025 }
                subtract_from_variable = { var:modifier@trade_opinion_factor = 0.025 }
                subtract_from_variable = { var:modifier@production_factory_start_efficiency_factor = 0.025 }
                subtract_from_variable = { var:modifier@production_speed_buildings_factor = 0.025 }
                subtract_from_variable = { var:modifier@industrial_capacity_factory = 0.025 }
                subtract_from_variable = { var:modifier@production_oil_factor = 0.025 }
                subtract_from_variable = { var:modifier@local_supplies = 0.025 }
            } else_if = {
                limit = { 
                    has_idea = mod_lockdown_outer
                    check_variable = {
                        covid_months > 20
                    }
                }

                add_to_variable = { var:modifier@local_resources_factor = 0.025 }
                add_to_variable = { var:modifier@trade_opinion_factor = 0.025 }
                add_to_variable = { var:modifier@production_factory_start_efficiency_factor = 0.025 }
                add_to_variable = { var:modifier@production_speed_buildings_factor = 0.025 }
                add_to_variable = { var:modifier@industrial_capacity_factory = 0.025 }
                add_to_variable = { var:modifier@production_oil_factor = 0.025 }
                add_to_variable = { var:modifier@local_supplies = 0.025 }
            }
            add_to_variable = { var:covid_months = 1 }
        }
    }
}
