# Game setup for all countries
on_actions = {
	on_startup = {
		effect = { # Setup variables 
			# supremacy gui
			set_variable = { global.total_countries = 0 }
			set_variable = { global.democratic_countries = 0 }
			set_variable = { global.fascism_countries = 0 }
			set_variable = { global.communism_countries = 0 }
			set_variable = { global.neutrality_countries = 0 }
			every_country = {
				if = {
					limit = { THIS = { has_government = democratic } }
					add_to_variable = { global.democratic_countries = 1 }
				}	
				if = {
					limit = { THIS = { has_government = fascism } }
					add_to_variable = { global.fascism_countries = 1 }
				}	
				if = {
					limit = { THIS = { has_government = communism } }
					add_to_variable = { global.communism_countries = 1 }
				}	
				if = {
					limit = { THIS = { has_government = neutrality } }
					add_to_variable = { global.neutrality_countries = 1 }
				}
				add_to_variable = { global.total_countries = 1 }
			}
		}
		effect = { # Playtester warning
			if = {
				limit = {
					is_debug = yes
				}
				every_country = {
					country_event = {
						id = non_war_events.3
						days = 1
					}
				}
			}
		}
	}
}
