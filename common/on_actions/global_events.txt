# This is for global events (the news).
on_actions = {
	on_startup = {
		effect = {
			set_variable = { months_passed = 0 }
		}
	}
	on_monthly = {
		effect = {
			add_to_variable = { months_passed = 1 }
		}
	}
	on_yearly = {
		effect = {
			add_to_variable = { ITA_years_since_last_election = 1 }
		}
	}

	# GLOBAL EVENTS
	on_monthly_GRE = { # Greek Civil War Trigger
		effect = {
			if = {
				limit = { 
					communism > 0.3 # KPD has 30% popularity
					check_variable = { 
						var = months_passed 
						value = 7
						comparison = greater_than_or_equals
					}
					NOT = { has_global_flag = GRE_civil_war } # one time only
					OR = {
						BUL = { has_government = communism }
						ROM = { has_government = communism }
					}
				}
				
				set_global_flag = GRE_civil_war
				start_civil_war = {
					ideology = communism
					ruling_party = neutrality
					states = { 184 731 1052 }
					capital = 731
					keep_political_leader = yes

					# For the Provisional Democratic Government
					set_cosmetic_tag = GRE_movement	# TAG = GRC
					load_oob = "GRC_uprising_1947"
				}

				YUG = {
					start_civil_war = {
						ideology = communism
						ruling_party = fascism
						states = { 106 970 803 802 108 }
						capital = 106
						keep_political_leader = no

						# For the Yugoslavian People's Republic
						set_cosmetic_tag = YUG_movement	# TAG = PRY
						retire_country_leader = yes
						promote_character = YUG_josip_broz_tito
						declare_war_on = {
							target = CRO
							type = take_core_state
						}
						load_oob = "PRY_uprising_1947"
					}
				}
				
				add_named_threat = {
					threat = 5
					name = international_events.1.t
				}
				news_event = {
					id = international_events.1
					days = 1
				}
			}
		}
	}

	on_daily_EGY = { # Egyptian Civil War Trigger
		effect = {
			if = {
				limit = { 
					fascism > neutrality
					check_variable = { 
						var = months_passed 
						value = 12
						comparison = greater_than_or_equals
					}
					NOT = { has_global_flag = EGY_civil_war }
					ITA = { has_completed_focus = focus_ITA_mediterranean_egypt_influence }
				}

				set_global_flag = EGY_civil_war
				start_civil_war = {
					ideology = fascism
					ruling_party = neutrality
					states = { 452 447 446 453 }
					capital = 447
					keep_political_leader = yes

					# For the Fascist State
					add_ideas = mod_ITA_trained_militias  
					set_cosmetic_tag = EGY_italy # TAG = IEG
					news_event = {
						id = international_events.2
						days = 1
					}
					ITA = { country_event = ITA_missiolinis_events.1_egypt }
					add_named_threat = {
						threat = 3
						name = international_events.2.t
					}
				}
			}
		}
	}

	on_daily_ITA = {
		effect = {
			if = {
				limit = {
					date > 1956.5.1
				}

				country_event = { # Mussolini Dead
					id = ITA_events.2
					random_days = 360
				}
			}
		}
	}
}
