				tooltip = IIA_elections_allowed_tt
				NOT = { has_country_flag = ITA_elections_allowed }
			}
		}

			hidden_trigger = { always = no }
		}
		is_good = yes

				set_variable = { ITA_years_since_last_election = 0 }
