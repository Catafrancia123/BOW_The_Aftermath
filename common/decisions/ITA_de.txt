# Add new missiolinis
ITA_missiolinis = {
	ITA_missiolini_conquer_egypt = {
		icon = GFX_decision_generic_operation
		allowed = { tag = ITA }
		visible = {
			ITA_missiolinis_system_available = yes
			has_completed_focus = focus_ITA_politics_egypt_influence
			IEG = { exists = yes } # Fascist Egypt
		}

		cancel = { # GOAL
			custom_trigger_tooltip = {
				tooltip = ITA_missiolinis.1_egypt.tt
				AND = {
					IEG = { 
						controls_province = 7011 
						controls_province = 12071
					} # Cairo and Luxor
					446 = { is_controlled_by = IEG } # Suez State
				}
			}
		}

		activation = {
			always = no
		}

		highlight_states = {
			highlight_state_targets = {
				state = 907 # Cairo
				state = 456 # Luxor
			}
			highlight_color_while_active = 3
			highlight_color_before_active = 2
		}

		cancel_if_not_visible = yes
		is_good = no # Tooltip purposes
		fire_only_once = yes
		selectable_mission = no
		days_mission_timeout = ROOT.ITA_missiolinis_egypt_days

		cancel_effect = {
			if = { # Mission Success!!!
				limit = {
					IEG = { 
						controls_province = 7011 
						controls_province = 12071
					} # Cairo and Luxor
					446 = { is_controlled_by = IEG } # Suez State
				}
				country_event = { id = ITA_missiolinis_events.2_egypt hours = 1 }
			}
			else = { # Mission Failed, we'll get them next time.
				# Prepare next mission
				if = {
					limit = { ITA_missiolinis_system_available = yes }
					ITA_missiolinis_manage_next_mission = yes
				}
				country_event = { id = ITA_missiolinis_events.3_egypt hours = 1 } # declare war on egypt
			}
		}

		timeout_effect = { # declare war!!
			hidden_effect = {
				#Prepare next mission
				if = {
					limit = {
						ITA_missiolinis_system_available = yes
					}
					ITA_missiolinis_manage_next_mission = yes
				}
				country_event = { id = ITA_missiolinis_events.3_egypt hours = 1 }
			}
		}

		complete_effect = { # Mission Success!!!
			hidden_effect = {
				#Prepare next mission
				if = {
					limit = {
						ITA_missiolinis_system_available = yes
					}
					ITA_missiolinis_manage_next_mission = yes
				}

				country_event = { id = ITA_missiolinis_events.2_egypt hours = 1 }
			}
		}
	}

	# More Military Factories
