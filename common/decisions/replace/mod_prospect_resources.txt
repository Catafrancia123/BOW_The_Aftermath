prospect_for_resources = {
	develop_sidamo_rubber_plantations = { #837
		icon = rubber
		highlight_states = {
			highlight_state_targets = {
				state = 837
			}
		}

		available = {
			set_temp_variable = { prospect_state = 837 }
			custom_trigger_tooltip = {
				tooltip = mod_prospect_trigger

				OR = { # any other country
					owns_state = 837
					controls_state = 837
					any_subject_country = { # AOI or the subject of any country
						owns_state = 837
						controls_state = 837
					}
				}
			}
			num_of_civilian_factories_available_for_projects > 5
			has_tech = excavation3
		}

		visible = {
			OR = { # any other country
				owns_state = 837
				controls_state = 837
				any_subject_country = { # AOI or the subject of any country
					owns_state = 837
					controls_state = 837
				}
			}
			837 = {
				NOT = { has_state_flag = sidamo_rubber_developed }
			}
		}

		fire_only_once = yes

		cost = 25
		days_remove = 180

		modifier = {
			civilian_factory_use = 5
			recruitable_population_factor = -0.25
		}

		ai_will_do = {
			base = 0.3
			modifier = {
				factor = 0
				OR = {
					AND = { #Ultimatum decisions are available and we should prio them
						original_tag = SOV
						SOV_we_want_to_send_ultimatums = yes
					}
					AND = { #Too low MP we need to save PP for the laws
						has_manpower < 10000
						NOT = { has_idea = scraping_the_barrel }
					}
				}
			}
			modifier = {  # Too low PP
				factor = 0.5
				has_political_power < 80
			}
			modifier = { #We have plenty of PP, avanti!
				factor = 3
				has_political_power > 200
			}
			modifier = { #Running out of the stuff, need more!
				factor = 5
				has_resources_in_country = {
					resource = rubber
					amount < 1
				}
			}
		}

		remove_effect = {
			# calculate amount of resources to master and puppet (if owner of state is puppet)
			set_temp_variable = { resource_get_base = 6 }
			if = {
				limit = {
					any_country = { # any other country
						any_subject_country = { # AOI or the subject of any country
							owns_state = 837
							controls_state = 837
						}
					}
				}
				set_temp_variable = { resource_get_master = 0.8 }
				multiply_temp_variable = { resource_get_master = resource_get_base } # master gets 80% of resources
				round_temp_variable = resource_get_master
				subtract_from_temp_variable = { resource_get_base = resource_get_master }
			}
			

			837 = {
				set_state_flag = sidamo_rubber_developed

				add_resource = {
					type = rubber
					amount = resource_get_base
				}
			}
		}
	}
}
