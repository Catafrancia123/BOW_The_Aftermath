# Generic Events
country_event = {
	# Generic event where a country accepts italian state demands (full)
	id = ITA_generic_events.1
	title = ITA_generic_events.1.t
	desc = ITA_generic_events.1.d
	picture = GFX_report_event_generic_italian_celebration
	is_triggered_only = yes
	option = {
		name = ITA_generic_events.1.op1
		# use array
		# FROM = the country that accepts
		# ROOT = ITA
		FROM = {
			ROOT = {
				for_each_loop = {
					array = ITA_requested_states
					value = v
					
					transfer_state = v
				}
				clear_array = ITA_requested_states
				set_variable = { ITA_requested_success = 1 }
			}
		}
		ROOT = {
			add_opinion_modifier = {
				target = FROM
				modifier = ITA_demands_granted
			}

			# flag thing
			if = {
				limit = { FROM = { tag = CRO } }
				set_country_flag = ITA_demand_dalmatia_success
			}
			if = {
				limit = { FROM = { tag = GRE } }
				set_country_flag = ITA_demand_corfu_success 
			}
		}
	}
}
